#include <stdio.h>
int main() {
    int total;
    if(scanf("%d",&total) !=1 || total <=0) 
    return 0;
    int blocks[total];
    for(int k = 0; k < total;k++) {
        if(scanf("%d", &blocks[k]) !=1)
         return 0;
    }
    int targetSum, length;
    if(scanf("%d %d", &targetSum, &length) !=2 || length <=0 || length >total) return 0;
    int validWays = 0;
    int running = 0;
    for(int p = 0; p < total; p++) {
        running += blocks[p];
        if(p >= length) running -= blocks[p - length];
        if(p >= length - 1 && running == targetSum)
         validWays++;
    }
    printf("%d\n", validWays);
    return 0;
}
