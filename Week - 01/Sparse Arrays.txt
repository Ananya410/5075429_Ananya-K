#include <stdio.h>
#include <string.h>
int main(void) {
    int n, q;
    if(scanf("%d", &n) != 1) 
    return 0;
char words[n][21];
    for(int i = 0; i < n; i++) {
        scanf("%20s", words[i]);
    }
if(scanf("%d", &q) != 1)
 return 0;
char search[21];
    for(int i = 0; i < q; i++) {
    scanf("%20s", search);
        int cnt = 0;
        for(int k = 0; k < n; k++) {
            if (strcmp(search, words[k]) == 0)
             cnt++;
        }
    printf("%d\n", cnt);
    }
 return 0;
}
