#include <stdio.h>
int main() {
    int n;
    if(scanf("%d", &n) !=1 || n<1)
     return 0;
    long diagonal1 = 0, diagonal2 = 0;
    for(int idx = 0; idx < n * n; idx++) {
        int value;
        if(scanf("%d",&value) !=1) 
        return 0;
        int row = idx / n;
        int column = idx % n;
        if(row == column) diagonal1 += value;
        if(row + column == n - 1) diagonal2 += value;
    }
    long answer = diagonal1 - diagonal2;
    if (answer < 0) answer = -answer;
    printf("%ld\n",answer);
    return 0;
}
