#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int freq[100] = {0};
    int val;
for(int i = 0;i<n;i++) {
        scanf("%d", &val);
        freq[val]++;
    }
for(int i = 0; i < 100; i++) {
        printf("%d", freq[i]);
        if (i < 99) 
        printf(" ");
    }
    printf("\n");
     return 0;
}
