#include <stdio.h>

int main(void) {
    char tstr[12];
if (scanf("%11s", tstr) != 1) 
return 0;
int h = (tstr[0] - '0') * 10 + (tstr[1] - '0');
int m = (tstr[3] - '0') * 10 + (tstr[4] - '0');
int s = (tstr[6] - '0') * 10 + (tstr[7] - '0');

    if(tstr[8] == 'A') {
        if(h == 12) 
        h = 0;
    } else {
        if(h!=12)
         h+=12;
    }
    printf("%02d:%02d:%02d\n", h, m, s);
    return 0;
}
