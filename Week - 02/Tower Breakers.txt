#include <stdio.h>
int towerBreakers(int towers, int height) {
    if(height == 1) 
    return 2;
    return(towers % 2 == 1) ? 1 : 2;
}
int main() {
    int t;
    if(scanf("%d", &t) !=1 || t<=0) 
    return 0;
    while(t--) {
        int n, m;
        if(scanf("%d %d", &n, &m) !=2 || n <=0 || m<=0) 
        return 0;
        int winner = towerBreakers(n,m);
        printf("%d\n",winner);
    }
   return 0;
}
