#include <stdio.h>
#include <stdlib.h>
int compare(const void *a, const void *b) {
    int x = *(const int*)a;
    int y = *(const int*)b;
    return x-y;
}
int maxMin(int k, int arr[],int n) {
    qsort(arr, n, sizeof(int),compare);
    int minUnfair = arr[k - 1]-arr[0];
    for (int i = 1;i <= n - k;i++) {
    int unfair = arr[i + k - 1] - arr[i];
    if(unfair < minUnfair)
     minUnfair = unfair;
    }
    return minUnfair;
}
int main() {
    int n, k;
if(scanf("%d", &n) !=1 || n <=0) 
return 0;
    if(scanf("%d", &k) != 1 || k <=0 || k>n) 
    return 0;
    int arr[n];
    for(int i = 0; i < n;i++) {
        if(scanf("%d", &arr[i]) !=1) 
        return 0;
    }
    int result = maxMin(k, arr, n);
    printf("%d\n",result);
    return 0;
}
