#include <stdio.h>
const char* balancedSums(int arr[],int n) {
    long long total = 0,leftSum = 0;
    for(int i = 0; i<n; i++) {
        total += arr[i];
    }
    for(int i = 0; i < n; i++) {
        long long rightSum = total - leftSum - arr[i];
        if(leftSum == rightSum) 
        return "YES";
        leftSum += arr[i];
    }
    return "NO";
}
int main() {
    int t;
    if(scanf("%d",&t) !=1 || t<=0) 
    return 0;
    while (t--) {
        int n;
        if(scanf("%d",&n) !=1 || n<=0) 
        return 0;
        int arr[n];
        for(int i=0; i<n;i++) {
            if(scanf("%d", &arr[i]) !=1)
             return 0;
        }
        printf("%s\n", balancedSums(arr, n));
    }
  return 0;
}
